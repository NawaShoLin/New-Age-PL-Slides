Quick Sort

Pseudocode of quick-sort on Wikipedia

function quicksort('array')
      if length('array') ‚â§ 1
          return 'array'  // an array of zero or one elements is already sorted
          
      select and remove a pivot value 'pivot' from 'array'
      create empty lists 'less' and 'greater'
      
      for each 'x' in 'array'
          if 'x' ‚â§ 'pivot' then append 'x' to 'less'
          else append 'x' to 'greater'
          
      return concatenate(quicksort('less'), 'pivot', quicksort('greater')) 
             // two recursive calls

http://en.wikipedia.org/wiki/Quicksort

Implementation in C

void swap(int* a, int* b){ int t = *a; *a = *b; *b = t; }

void QSort(int* ary, int left, int right){
    if(left >= right){ return; }
    int l = left, r = right + 1;    
    while(1){
        while(l + 1 < right && ary[++l] < ary[left]);
        while(r - 1 >= left && ary[--r] > ary[left]);
        if(l >= r){ break; }
        swap(&ary[l], &ary[r]);
        ++l; --r;
    }
    swap(&ary[r], &ary[left]);

    Qsort(ary, left, r-1);
    Qsort(ary, r+1, right);
}

Implementation in Python

def quick_sort(ary):
    if not ary : return [] # end if array is empty

    left  = quick_sort([ x for x in ary[1:] if x <  ary[0] ])
    right = quick_sort([ x for x in ary[1:] if x >= ary[0] ])
    return left + ary[0:1] + right
