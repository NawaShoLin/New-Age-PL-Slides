(defun star-line(n)
  (when (> n 0)
    (format t "*")
    (star-line (- n 1))))

(defun stars(n end)
  (when (<= n end)
    (star-line n)
    (format t "~%")
    (stars (+ n 1) end)))

int* Merge(int a[], int b[], int a_size, int b_size){
    int* new_ary = malloc( (a_size + b_size) * sizeof(int) );
    int  i;
    
    for(i = 0 ; i < a_size ; ++i)
        new_ary[i] = a[i];

C swap function

void swap(int* a, int* b){int temp = *a; *a = *b; *b = temp;}

int main(){
    int a, b;
    swap(&a, &b);
}

C++ swap function

void swap(int& a, int& b){int temp = a; a=b; b=temp;}

int main(){
    int a, b;
    swap(a,b);
}

C++ swap (template)

template<class T>
void swap(T& a, T& b){T temp = a; a = b; b = temp;}

int main(){
    int a, b;
    float c, d;
    
    swap(a,b);
    swap(c,d);
}
